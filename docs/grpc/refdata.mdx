## Example: Fetching Market RefData

```python python
import grpc
import common_pb2
import market_data_pb2_grpc

def fetch_refdata():
    channel = grpc.secure_channel('mds.pfex.io:443', grpc.ssl_channel_credentials())
    stub = market_data_pb2_grpc.MarketDataServiceStub(channel)
    response = stub.GetRefData(common_pb2.Empty())
    print(response)

if __name__ == '__main__':
    fetch_refdata()
````

```java java
import io.grpc.ManagedChannel;
import io.grpc.ManagedChannelBuilder;
import io.grpc.Metadata;
import io.grpc.stub.MetadataUtils;
import common.Empty;
import market_data.MarketDataServiceGrpc;
import market_data.RefData;

public class RefDataFetcher {
    public static void main(String[] args) {
        ManagedChannel channel = ManagedChannelBuilder.forAddress("mds.pfex.io", 443)
            .useTransportSecurity()
            .build();

        MarketDataServiceGrpc.MarketDataServiceBlockingStub stub =
            MarketDataServiceGrpc.newBlockingStub(channel);

        RefData response = stub.getRefData(Empty.newBuilder().build());
        System.out.println(response);

        channel.shutdown();
    }
}
````

```go go
package main

import (
    "context"
    "fmt"
    "log"

    "google.golang.org/grpc"
    "google.golang.org/grpc/credentials"
    pbCommon "path/to/proto/common"
    pb "path/to/proto/market_data"
)

func main() {
    creds := credentials.NewClientTLSFromCert(nil, "")
    conn, err := grpc.Dial("mds.pfex.io:443", grpc.WithTransportCredentials(creds))
    if err != nil {
        log.Fatalf("did not connect: %v", err)
    }
    defer conn.Close()

    client := pb.NewMarketDataServiceClient(conn)

    resp, err := client.GetRefData(context.Background(), &pbCommon.Empty{})
    if err != nil {
        log.Fatalf("error on GetRefData: %v", err)
    }

    fmt.Println(resp)
}
```

```javascript node
const grpc = require("@grpc/grpc-js");
const protoLoader = require("@grpc/proto-loader");

// Load proto definitions
const commonDef = protoLoader.loadSync("common.proto", {
  keepCase: true,
  longs: String,
  enums: String,
  defaults: true,
  oneofs: true,
});
const marketDef = protoLoader.loadSync("market_data.proto", {
  keepCase: true,
  longs: String,
  enums: String,
  defaults: true,
  oneofs: true,
});
const commonProto = grpc.loadPackageDefinition(commonDef).common;
const marketDataProto = grpc.loadPackageDefinition(marketDef).market_data;

function fetchRefData() {
  const client = new marketDataProto.MarketDataService(
    "mds.pfex.io:443",
    grpc.credentials.createSsl()
  );

  client.getRefData({}, (err, response) => {
    if (err) {
      console.error(err);
      return;
    }
    console.log(response);
  });
}

fetchRefData();
```
