[StreamOrders RPC](https://github.com/QFEX-org/proto/blob/main/port.proto#L9) provides a bidirectional streaming interface for sending Add, Modify, and Cancel order requests and receiving responses.

## Example Requests 

### [Add Order](https://github.com/QFEX-org/proto/blob/main/port.proto#L27)

<CodeGroup>
```python python

import port_pb2

import common_pb2
````

```java java
import io.grpc.ManagedChannel;
import io.grpc.ManagedChannelBuilder;
import io.grpc.Metadata;
import io.grpc.stub.MetadataUtils;
import io.grpc.stub.StreamObserver;
import com.qfex.port.PortServiceGrpc;

import com.qfex.port.ModifyOrderRequest;
import com.qfex.port.CancelOrderRequest;

public class OrderStreamer {
    public static void main(String[] args) throws InterruptedException {
        String apiKeyValue = System.getenv("QFEX_API_KEY");
        if (apiKeyValue == null || apiKeyValue.isEmpty()) {
            System.err.println("ERROR: QFEX_API_KEY not set");
            System.exit(1);
        }

        ManagedChannel channel = ManagedChannelBuilder.forAddress("trade.pfex.io", 443)
                .useTransportSecurity()

                new StreamObserver<OrderResponse>() {
                    @Override
                    public void onNext(OrderResponse value) {
                        System.out.println("Received: " + value);
                    }
                    @Override
                    public void onError(Throwable t) {
                        t.printStackTrace();
                    }
                    @Override
                    public void onCompleted() {
                        System.out.println("Stream completed");
                    }
                }
        );

        // Send add order

                .setSymbol("SP500-USD")
                .setSide(OrderDirection.BUY)

                .setQuantity(0.1)

                OrdersStreamRequest.newBuilder().setAddOrder(add).build()
        );

        // Mark end of requests
        requestObserver.onCompleted();

        // Wait for responses
        Thread.sleep(1000);
        channel.shutdown();
    }
}
````

```go go
package main

import (
    "context"
    "fmt"
    "io"
    "log"

    "google.golang.org/grpc"
    "google.golang.org/grpc/credentials"
    "google.golang.org/grpc/metadata"
    pb "path/to/proto/port"
    pbCommon "path/to/proto/common"
)

func main() {
    creds := credentials.NewClientTLSFromCert(nil, "")
    conn, err := grpc.Dial("trade.pfex.io:443", grpc.WithTransportCredentials(creds))
    if err != nil {
        log.Fatalf("did not connect: %v", err)
    }
    defer conn.Close()

    client := pb.NewPortServiceClient(conn)
    ctx := metadata.AppendToOutgoingContext(context.Background(), "api-key", API_KEY)

    stream, err := client.StreamOrders(ctx)
    if err != nil {
        log.Fatalf("error on StreamOrders: %v", err)
    }

    // Send commands concurrently
    go func() {
        // Add a limit buy order
        stream.Send(&pb.OrdersStreamRequest{
            AddOrder: &pb.AddOrderRequest{
                Symbol:         "SP500-USD",
                Side:           pbCommon.OrderDirection_BUY,
                Type:           pbCommon.OrderType_LIMIT,
                TimeInForce:    pbCommon.OrderTimeInForce_GTC,
                Price:          560.50,
                Size:           0.1,
                ClientOrderId:  "order-id-123",
            },
        })
        stream.CloseSend()
    }()

    // Receive responses
    for {
        resp, err := stream.Recv()
        if err == io.EOF {
            break
        }
        if err != nil {
            log.Fatalf("stream error: %v", err)
        }
        fmt.Println("Response:", resp)
    }
}
```

### [Modify Order](https://github.com/QFEX-org/proto/blob/main/port.proto#L46)
<Note>
  You can only modify orders by **`orderId`**. 
  
  This is to prevent modifications of inflight orders. If you must modify an order before receiving an ACK,
  cancel the order and re-add it.

  You cannot modify an order that has been partially filled.
</Note>

<CodeGroup>
```python python
import os

import port_pb2
import port_pb2_grpc
import common_pb2

    metadata = (('api-key', API_KEY),)
    channel = grpc.secure_channel('trade.pfex.io:443', grpc.ssl_channel_credentials())
    stub = port_pb2_grpc.PortServiceStub(channel)

    def request_generator():
        # Modify an existing limit order
        yield port_pb2.OrdersStreamRequest(
            modify_order=port_pb2.ModifyOrderRequest(
                order_id="order-12345",
                symbol="SP500-USD",
                price=4210.00,
                size=2.0
            )
        )

    responses = stub.StreamOrders(request_generator(), metadata=metadata)
    for resp in responses:
        print("Response:", resp)

if __name__ == '__main__':
    modify_order()
````

### [Cancel Order](https://github.com/QFEX-org/proto/blob/main/port.proto#L38)
<Note>
  You can cancel an order in two ways: 
  
  1. **By `order_id`**: Cancels the single
  order matching that unique identifier. 
  
  2. **By `client_order_id`**: Cancels _all_ orders 
  for your account that share the same client_order_id. Uniqueness is
  not enforced, so multiple orders may be cancelled at once.
</Note>

<CodeGroup>
```python python
import os

import grpc
import port_pb2
import port_pb2_grpc
import common_pb2

API_KEY = os.getenv("QFEX_API_KEY")

def cancel_orders():
    metadata = (('api-key', API_KEY),)
    channel = grpc.secure_channel('trade.pfex.io:443', grpc.ssl_channel_credentials())
    stub = port_pb2_grpc.PortServiceStub(channel)

    # Cancel by orderId
    stub.CancellableStream(common_pb2.Empty(), metadata=metadata)  # initialize bidi if needed
    responses = stub.StreamOrders(
        iter([
            port_pb2.OrdersStreamRequest(
                cancelOrder=port_pb2.CancelOrderRequest(
                    orderId="abc12345-12345-12345-12345",
                    symbol="SP500-USD",
                )
            )
        ]),
        metadata=metadata
    )
    for resp in responses:
        print(resp)

    # Cancel by clientOrderId
    responses = stub.StreamOrders(
        iter([
            port_pb2.OrdersStreamRequest(
                cancelOrder=port_pb2.CancelOrderRequest(
                    clientOrderId="my-client-id-001",
                    symbol="SP500-USD",
                )
            )
        ]),
        metadata=metadata
    )
    for resp in responses:
        print(resp)

if __name__ == '__main__':
    cancel_orders()
````
